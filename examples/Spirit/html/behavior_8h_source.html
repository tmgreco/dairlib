145 void getModeSequenceHelper(dairlib::ModeSequenceHelper& msh){
146 int counter=0;
147 for (std::string mode_name : mode_vector){
148 Eigen::Matrix<bool,1,4> contact_bool;
149 if (mode_name=="stance") contact_bool<< true, true, true, true;
150 else if (mode_name=="flight") contact_bool<< false, false, false, false;
151 else if (mode_name=="rear_stance") contact_bool<<false, true, false, true;
152 else if (mode_name=="front_stance") contact_bool<< true, false, true ,false;
153 else std::cout<<"Wrong mode name!"<<std::endl;
154 msh.addMode( // Stance
155 contact_bool, // contact bools
156 num_knot_points[counter], // number of knot points in the collocation
157 normal_vector[counter], // normal
158 offset_vector[counter], // world offset
159 mu, //friction
160 minT_vector[counter], // minimum time
161 maxT_vector[counter] // maximum time
162 );
163
164 counter++;
165 }

178 if (ipopt) {
179 // Ipopt settings adapted from CaSaDi and FROST
180 auto id = drake::solvers::IpoptSolver::id();
181 trajopt.SetSolverOption(id, "tol", tol);
182 trajopt.SetSolverOption(id, "dual_inf_tol", tol);
183 trajopt.SetSolverOption(id, "constr_viol_tol", tol);
184 trajopt.SetSolverOption(id, "compl_inf_tol", tol);
185 trajopt.SetSolverOption(id, "max_iter", 1000000);
186 trajopt.SetSolverOption(id, "nlp_lower_bound_inf", -1e6);
187 trajopt.SetSolverOption(id, "nlp_upper_bound_inf", 1e6);
188 trajopt.SetSolverOption(id, "print_timing_statistics", "yes");
189 trajopt.SetSolverOption(id, "print_level", 5);
190
191 // Set to ignore overall tolerance/dual infeasibility, but terminate when
192 // primal feasible and objective fails to increase over 5 iterations.
193 trajopt.SetSolverOption(id, "acceptable_compl_inf_tol", tol);
194 trajopt.SetSolverOption(id, "acceptable_constr_viol_tol", tol);
195 trajopt.SetSolverOption(id, "acceptable_obj_change_tol", tol);
196 trajopt.SetSolverOption(id, "acceptable_tol", tol);
197 trajopt.SetSolverOption(id, "acceptable_iter", 5);
198 } else {
199 // Set up Trajectory Optimization options
200 trajopt.SetSolverOption(drake::solvers::SnoptSolver::id(),
201 "Print file", "../snopt.out");
202 trajopt.SetSolverOption(drake::solvers::SnoptSolver::id(),
203 "Major iterations limit", 100000);
204 trajopt.SetSolverOption(drake::solvers::SnoptSolver::id(), "Iterations limit", 100000);
205 trajopt.SetSolverOption(drake::solvers::SnoptSolver::id(),
206 "Major optimality tolerance",
207 tol); // target optimality
208 trajopt.SetSolverOption(drake::solvers::SnoptSolver::id(), "Major feasibility tolerance", tol);
209 trajopt.SetSolverOption(drake::solvers::SnoptSolver::id(), "Verify level",
210 0); // 0

215 void initTrajectory(dairlib::systems::trajectory_optimization::Dircon<Y>& trajopt,
216 DirconModeSequence<Y>& sequence){
217 if (file_name_in.empty()){
218 trajopt.drake::systems::trajectory_optimization::MultipleShooting::

220 for (int j = 0; j < sequence.num_modes(); j++) {
